task Fire
{
	let ang = 0;
	let bit = 1;
	while(!Obj_IsDeleted(objEnemy))
	{
		
		//SE_PlayLen(chime1,60);
		
		HecatiaChunk(ang,bit);
		
		wait(6);
		
		ang+=(360/7 + 9)*bit;
	}
}
task HecatiaChunk(ang,bit)
{
	let d = 3;
	let ds = 20;
	ascent(i in 0..6)
	{
		SE_PlayLen(shot1,60);
		ascent(o in 0..3)
		{
			let ang2 = ang - 3*o*bit;
			let delay = ds+d+d*o;
			let obj = ShotA1(ObjMove_GetX(objEnemy)+60*cos(ang2),ObjMove_GetY(objEnemy)+60*sin(ang2),2,ang2,425,418,delay,3,BLEND_ADD_ARGB,false);
			ObjRender_SetBlendType(obj,BLEND_ADD_ARGB);
			Curvy(obj,ang2,delay);
			
		}
		ang+=8*bit;
		wait(2);
	}
}
task Curvy(obj,ang,delay)
{
	wait(delay);
	let cc = 0;
	while(!Obj_IsDeleted(obj))
	{
		ObjMove_SetAngle(obj,ang+45*sin(cc));
		cc+=2;
		yield;
	}
}

    }
}
task end

} 
task move
