task Fire
{
	let laserAng = 0;
	let timerSpin = 40*60;
	let laserSpinSpeed = 360/(timerSpin);
	ascent(i in 0..8)
	{
		SpinningLaser(360/8*i,i);
	}
	while(!Obj_IsDeleted(objEnemy))
	{
		laserAng+=laserSpinSpeed;
		if(timerSpin > 15*60)
		{
			timerSpin-=(25*60) / (50*60);
		}
		laserSpinSpeed = 360/(timerSpin);
		yield;
	}
	task SpinningLaser(angP,spinID)
	{
		let obj = CreateStraightLaserA1(ObjMove_GetX(objEnemy),ObjMove_GetY(objEnemy),angP,1000,10,30,1,999*60);
		ObjStLaser_SetSource(obj,false);
		let cc = 0;
		let type = spinID%2;
		let gap = 5;
		while(!Obj_IsDeleted(objEnemy))
		{
			let x = ObjMove_GetX(obj);
			let y = ObjMove_GetY(obj);
			let dis = Get_Distance(x,y,GetPlayerX,GetPlayerY);
			let newAng = laserAng + angP;
			ObjStLaser_SetAngle(obj,newAng);
			if(cc%6==0)
			{
				let newX = x+dis*cos(newAng);
				let newY = y+dis*sin(newAng);
				if(newX < gap) newX = gap;
				if(newX > GetStgFrameWidth-gap) newX = GetStgFrameWidth-gap;
				if(newY < gap) newY = gap;
				if(newY > GetStgFrameHeight-gap) newY = GetStgFrameHeight-gap;
				
				if(spinID==0)
				{
					SE_PlayLen(shot1,60);
				}
				let obj2 = ShotA1(newX,newY,0,newAng+90,[139,141][type],[418,419][type],5,2,BLEND_ALPHA,false);
				ObjShot_SetDeleteFrameA5(obj2,((360/8)/laserSpinSpeed)-5,1);
			}
			cc++;
			yield;
		}
	}
}
